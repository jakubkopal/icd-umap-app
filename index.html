<body>


    <h2>Delphi AI: ICD-10 Embedding Visualization</h2>

    <div id="plot"></div>

    <div id="info">Click two points to compare cosine similarity.</div>


    <script>

        let allData = [];

        let selection = [];


        d3.csv("embeddings.csv").then(data => {

            allData = data;

            renderPlot(data);

        });


        function renderPlot(data) {

            const trace = {

                x: data.map(d => d.umap_x),

                y: data.map(d => d.umap_y),

                mode: 'markers',

                type: 'scatter',

                text: data.map(d => d.name),

                marker: { color: data.map(d => d.color), size: 8 }

            };


            Plotly.newPlot('plot', [trace], { hovermode: 'closest' });


            document.getElementById('plot').on('plotly_click', function(data){

                const pointIndex = data.points[0].pointNumber;

                const pointData = allData[pointIndex];

                

                selection.push(pointData);

                if (selection.length > 2) selection.shift();


                updateUI();

            });

        }


        function cosineSimilarity(vecA, vecB) {

            let dotProduct = 0;

            let normA = 0;

            let normB = 0;

            for (let i = 0; i < 120; i++) {

                let a = parseFloat(vecA[`emb_${i}`]);

                let b = parseFloat(vecB[`emb_${i}`]);

                dotProduct += a * b;

                normA += a * a;

                normB += b * b;

            }

            return dotProduct / (Math.sqrt(normA) * Math.sqrt(normB));

        }


        function updateUI() {

            const infoDiv = document.getElementById('info');

            if (selection.length === 2) {

                const sim = cosineSimilarity(selection[0], selection[1]);

                infoDiv.innerHTML = `

                    <strong>Point 1:</strong> ${selection[0].name} (${selection[0].icd3})<br>

                    <strong>Point 2:</strong> ${selection[1].name} (${selection[1].icd3})<br>

                    <strong>Cosine Similarity:</strong> ${sim.toFixed(4)}

                `;

            } else {

                infoDiv.innerHTML = `Selected: ${selection[0].name}. Click another point...`;

            }

        }

    </script>
